SublimeSlideshow

CSS3を利用したフルスクリーンスライドショーjQueryプラグイン

元ネタ 
    $(function(){
        $.sublime_slideshow({
            src:[
            {url:"images/1.jpg",title:"星空"},
            {url:"images/2.jpg"},
            {url:"images/3.jpg",title:"月夜"}
            ],
            duration:   7,
            fade:       1,
            scaling:    1.2,
            rotating:   2,
            overlay:    "images/pattern.png"
        });
    });
